-- === [ ฟังก์ชันระบบความปลอดภัย (Noclip) ] ===
local noclip = nil
local function enableNoclip()
if noclip then return end
noclip = RunService.Stepped:Connect(function()
if player.Character then
for _, part in pairs(player.Character:GetDescendants()) do
if part:IsA("BasePart") then part.CanCollide = false end
end
end
end)
end
local function disableNoclip()
if noclip then noclip:Disconnect() noclip = nil end
end


-- === [ 💥 ฟังก์ชัน Dupe และกดเก็บของแบบรัวๆ 💥 ] ===
local function executeDupeAndCollect(prox)
-- ยิงสแปมคำสั่งกดรัวๆ (Dupe Network Spam)
for i = 1, 15 do
task.spawn(function()
if prox and prox.Parent then
if fireproximityprompt then
fireproximityprompt(prox)
else
prox:InputHoldBegin()
prox:InputHoldEnd()
end
end
end)
end

-- ลูปกดย้ำจนกว่าของจะหายไป (พร้อม Timeout กันค้าง)
local timeout = 0
repeat
if prox and prox.Parent then
if fireproximityprompt then
fireproximityprompt(prox)
else
prox:InputHoldBegin()
task.wait(0.05)
prox:InputHoldEnd()
end
end
task.wait()
timeout = timeout + 1
until not prox.Parent or timeout > 40
end


-- === [ ลูประบบ Auto Herbs (อัปเกรดให้ใช้การ Dupe) ] ===
task.spawn(function()
while true do
task.wait(0.1)
if isAutoHerbs then
pcall(function()
local herbsFolder = Workspace:FindFirstChild("Herbs")
if herbsFolder then
for _, obj in pairs(herbsFolder:GetChildren()) do
if not isAutoHerbs then break end

-- ตรวจสอบชื่อว่าตรงกับที่เลือกใน UI หรือไม่ (เหมือนเดิมเป๊ะ)
if selectedHerb == "All" or obj.Name == selectedHerb then
local prompt = obj:IsA("ProximityPrompt") and obj or obj:FindFirstChildOfClass("ProximityPrompt") or obj:FindFirstChild("ProximityPrompt", true)
if prompt then
local char = player.Character
local root = char and char:FindFirstChild("HumanoidRootPart")
local targetPart = obj:IsA("BasePart") and obj or obj:FindFirstChildOfClass("BasePart") or obj:FindFirstChildWhichIsA("BasePart", true)
if not targetPart and obj:IsA("Model") then targetPart = obj.PrimaryPart end

if root and targetPart then
local oldCFrame = root.CFrame -- จำพิกัดก่อนวาร์ป

enableNoclip()
root.CFrame = targetPart:GetPivot() + Vector3.new(0, 1, 0)
task.wait(0.15)

-- เรียกใช้ฟังก์ชัน Dupe ที่เพิ่งสร้างใหม่!
executeDupeAndCollect(prompt)

disableNoclip()

-- วาร์ปกลับจุดเดิมหลังเสร็จสิ้นการดูด/ดูป
root.CFrame = oldCFrame
task.wait(0.1)
end
end
end
end
end
end)
end
end
end)


-- === [ ระบบค้นหาและโจมตี Kill Quest ] ===
local function findTargetNPC()
for _, v in pairs(Workspace:GetDescendants()) do
if v:IsA("Model") and v.Name == selectedNPC then
local hum = v:FindFirstChildOfClass("Humanoid")
local hrp = v:FindFirstChild("HumanoidRootPart")
if hum and hrp and hum.Health > 0 then return v end
end
end
return nil
end

task.spawn(function()
while true do
task.wait(0.1)
if isKillQuest then
if not targetNPC or not targetNPC:FindFirstChild("Humanoid") or targetNPC.Humanoid.Health <= 0 or not targetNPC:FindFirstChild("HumanoidRootPart") then
targetNPC = findTargetNPC()
end
else
targetNPC = nil
end
end
end)

task.spawn(function()
while true do
task.wait(0.01)
if isKillQuest and targetNPC and targetNPC:FindFirstChild("HumanoidRootPart") then
pcall(function()
local char = player.Character
local root = char and char:FindFirstChild("HumanoidRootPart")
if root then
enableNoclip()
root.CFrame = targetNPC.HumanoidRootPart.CFrame * CFrame.new(0, 1.5, 2.2)
local tool = player.Backpack:FindFirstOfClass("Tool") or char:FindFirstOfClass("Tool")
if tool then tool.Parent = char tool:Activate() end
end
end)
else
if not isAutoHerbs then disableNoclip() end
end
end
end)


-- === [ ระบบสลับสไตล์สีปุ่มเมื่อ เปิด/ปิด (ธีม ดำ-ม่วง) ] ===
local function toggleStyle(btn, state)
if state then
btn.BackgroundColor3 = Color3.fromRGB(130, 45, 210)
btn.TextColor3 = Color3.fromRGB(255, 255, 255)
 else
btn.BackgroundColor3 = Color3.fromRGB(30, 25, 40)
 btn.TextColor3 = Color3.fromRGB(180, 150, 210)
 end
end

HerbBtn.MouseButton1Click:Connect(function()
isAutoHerbs = not isAutoHerbs
HerbBtn.Text = "Auto Herbs: " .. (isAutoHerbs and "ON" or "OFF")
toggleStyle(HerbBtn, isAutoHerbs)
if not isAutoHerbs then disableNoclip() end
end)

HerbDropdown.Visible = not HerbDropdown.Visible
NpcDropdown.Visible = false
end)

KillBtn.MouseButton1Click:Connect(function()
isKillQuest = not isKillQuest
KillBtn.Text = "Kill Quest: " .. (isKillQuest and "ON" or "OFF")

if not isKillQuest then disableNoclip() targetNPC = nil end
end)

NpcDropdown.Visible = not NpcDropdown.Visible
HerbDropdown.Visible = false
end)

-- ลากเมนูบนหน้าจอมือถือ
local dragging, dragInput, dragStart, startPos
MainFrame.InputBegan:Connect(function(input)
if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
dragging = true dragStart = input.Position startPos = MainFrame.Position
input.Changed:Connect(function() if input.UserInputState == Enum.UserInputState.End then dragging = false end end)
end
end)
UserInputService.InputChanged:Connect(function(input)
if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
local delta = input.Position - dragStart
MainFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
end
end)
